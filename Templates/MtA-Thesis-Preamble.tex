% The package ebgaramond uses the Octavio Pardo version, which does not have contextual alternates
% I downloaded the George Duffner version
% This has the contextual alternates and many more features
% I placed the fonts in a folder called font/
% Then, I used this code to manually set the main font
% +calt ensures that the Q.long is used
\setmainfont{EBGaramond}[
  Path=../Templates/EBGaramond-0.016/otf/,
  Extension=.otf,
  UprightFont=*12-Regular,
  ItalicFont=*12-Italic,
  RawFeature=+calt,
]

% Define a new style that allows for a bigger size than \HUGE

% Package to style the section headers
\usepackage{sectsty}

% Style the section headers to have some space before and after and use a larger font size
% https://tex.stackexchange.com/questions/381256/how-to-adjust-the-title-spacing-using-sectsty-package

% was originally       {\newpage\vspace*{0.25in}\normalfont\Huge\bfseries\SS@sectfont}}
\makeatletter
\renewcommand\section{\@startsection {section}{1}{\z@}%
      {-7ex \@plus -1ex \@minus -.2ex}%
      {7ex \@plus.2ex}%
      {\newpage\vspace*{-0.2in}\normalfont\BIG\SS@sectfont}}
\renewcommand\subsection{\@startsection{subsection}{2}{\z@}%
      {-3.5ex\@plus -1ex \@minus -.2ex}%
      {3.5ex \@plus .2ex}%
      {\normalfont\LARGE\itshape\SS@subsectfont}}
\makeatother
